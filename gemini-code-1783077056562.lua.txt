-- [[ CONFIGURATION MANAGEMENT ]]
function LoadSettings()
	if readfile and writefile and isfile and isfolder then
		if not isfolder("Attack Hub") then makefolder("Attack Hub") end
		if not isfolder("Attack Hub/Blox Fruits/") then makefolder("Attack Hub/Blox Fruits/") end
		local filePath = "Attack Hub/Blox Fruits/" .. game.Players.LocalPlayer.Name .. ".json"
		if not isfile(filePath) then
			writefile(filePath, game:GetService("HttpService"):JSONEncode(_G.Settings))
		else
			local Decode = game:GetService("HttpService"):JSONDecode(readfile(filePath))
			for i, v in pairs(Decode) do _G.Settings[i] = v end
		end
	else
		return warn("Status : Undetected Executor")
	end
end

function SaveSettings()
	if readfile and writefile and isfile and isfolder then
		local filePath = "Attack Hub/Blox Fruits/" .. game.Players.LocalPlayer.Name .. ".json"
		local Array = {}
		for i, v in pairs(_G.Settings) do Array[i] = v end
		writefile(filePath, game:GetService("HttpService"):JSONEncode(Array))
	end
end

spawn(function()
	pcall(function()
		while wait(1) do SaveSettings() end
	end)
end)
